    public function index(): View

    /**
     * Display the specified article based on the provided slug.
     *
     * @param string $slug
     * @return View|Response
     */
    public function showArticle(string $slug): View|Response
    {
        $article = $this->articleService->getPublishedArticleBySlug($slug);

        if (! $article instanceof Article) {
            abort(404);
        }

        $this->meta->setTitle($article->meta_title)
            ->setKeywords($article->meta_keyword)
            ->setDescription($article->meta_description);

        $article->load([
            'category'       => static function ($query): void {
                $query->select('id', 'name', 'slug');
            },
            'author'         => static function ($query): void {
                $query->select('id', 'name', 'email');
            },
            'author.profile' => static function ($query): void {
                $query->select('user_id', 'avatar_url', 'bio', 'position');
            },
        ]);

        $this->setArticleAttributes($article);

        $schema = Schema::article()
            ->name($article->title)
            ->headline($article->title)
            ->description(strip_tags($article->meta_description))
            ->author(Schema::person()->name($article->author_name))
            ->datePublished($article->published_at)
            ->dateModified($article->updated_at)
            ->image(Schema::imageObject()->url($article->image))
            ->publisher(
                Schema::organization()
                    ->name(config('app.name'))
                    ->logo(Schema::imageObject()->url(asset('logo_thegioiphp.png')))
            );

        $breadcrumbList = Schema::breadcrumbList()
            ->itemListElement([
                Schema::listItem()
                    ->position(1)
                    ->name(trans('homepage'))
                    ->item(route('frontend.index')),
                Schema::listItem()
                    ->position(2)
                    ->name($article->category_name)
                    ->item($article->category_url),
                Schema::listItem()
                    ->position(3)
                    ->name($article->title)
                    ->item($article->url),
            ]);

        $tocResult = $this->tocService->generateToc($article->content);

        $limit = 8;

        $lengthAwarePaginator = $this->getLatestArticlesWithPageNo(
            request()->merge(['limit' => $limit])
        );

        $banners = $this->bannerService->getBannersByPosition(
            BannerPositionType::SUBPAGE->value
        );

        $view = $article->is_tutorial
            ? 'frontend::blog.article_detail_for_tutorial'
            : 'frontend::blog.article_detail';

        return view(
            $view,
            [
                'article'          => $article,
                'content'          => Arr::get($tocResult, 'updated_html', ''),
                'toc'              => Arr::get($tocResult, 'toc', ''),
                'recentPosts'      => $lengthAwarePaginator,
                'schema'           => $schema,
                'breadcrumbSchema' => $breadcrumbList,
                'banners'          => $banners,
            ]
        );
    }

    /**
     * Display the specified category page based on the provided slug.
     *
     * @param Request $request
     * @param string $slug
     * @return View|Response
     */
    public function showCategory(Request $request, string $slug): View|Response
    {
        $category = $this->categoryService->getCategoryBySlug($slug);

        if (! $category instanceof Category) {
            abort(404);
        }

        $this->meta->setTitle($category->meta_title)
            ->setKeywords($category->meta_keyword)
            ->setDescription($category->meta_description);

        $request->merge(['category_id' => $category->id]);

        $lengthAwarePaginator = $this->articleService->getFrontLatestArticles($request);

        foreach ($lengthAwarePaginator as $article) {
            $this->setArticleAttributes($article, $category);
        }

        $limit       = 8;
        $recentPosts = $this->getLatestArticlesWithPageNo(
            request()->merge(['limit' => $limit])
        );

        $banners = $this->bannerService->getBannersByPosition(
            BannerPositionType::SUBPAGE->value
        );

        return view(
            'frontend::blog.category',
            [
                'category'    => $category,
                'articles'    => $lengthAwarePaginator,
                'recentPosts' => $recentPosts,
                'banners'     => $banners,
            ]
        );
    }

    /**
     * Set additional attributes for an article instance.
     *
     * @param Article $article
     * @param Category|null $category
     * @return void
     */
    protected function setArticleAttributes(Article $article, ?Category $category = null): void
    {
        $category = $category instanceof Category ? $category : $article->category;

        $category_name = trans('common.undefined');
        $category_slug = '';
        $category_url  = '';

        if ($category) {
            $category_name = $category->name;
            $category_slug = $category->slug;
        }

        $category_url = empty($category_slug)
            ? '#'
            : route('frontend.blog.category.detail', ['slug' => ($category->slug ?? '')]);

        $article->setAttribute(
            'url',
            route('frontend.blog.article.detail', ['slug' => $article->slug])
        );

        $article->setAttribute(
            'image',
            method_exists($article, 'getImageUrl') ? $article->getImageUrl() : $article->image
        );

        $article->setAttribute('category_name', $category_name);
        $article->setAttribute('category_url', $category_url);
        $article->setAttribute('published_at', $article->published_at ?? null);
        $article->setAttribute('author_name', $article->author->name ?? null);
        $article->setAttribute('author_avatar', $article->author->profile->avatar_url ?? null);
    }

    /**
     * Retrieves the latest articles along with the current page number from the request.
     *
     * @param Request $request
     * @return LengthAwarePaginator
     */
    protected function getLatestArticlesWithPageNo(Request $request): LengthAwarePaginator
    {
        $lengthAwarePaginator = $this->articleService->getFrontLatestArticles($request);

        $lengthAwarePaginator->load([
            'category' => static function ($query): void {
                $query->select('id', 'name', 'slug');
            },
            'author'   => static function ($query): void {
                $query->select('id', 'name');
            },
        ]);

        foreach ($lengthAwarePaginator as $article) {
            $this->setArticleAttributes($article);
        }

        return $lengthAwarePaginator;
    }
}
